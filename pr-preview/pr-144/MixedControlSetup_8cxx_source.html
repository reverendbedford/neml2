33MixedControlSetup::expected_options()
34{
35 OptionSet options = Model::expected_options();
36 options.doc() =
37 "Object to setup a model for mixed stress/strain control. Copies the values of the "
38 "fixed_values (the input strain or stress) and the mixed_state (the conjugate stress or "

41 options.set<VariableName>("control") = VariableName("forces", "control");
42 options.set("control").doc() =
43 "The name of the control signal. Values less than the threshold are "

50 options.set("mixed_state").doc() = "The name of the mixed state tensor. This holds the conjugate "
51 "values to those being controlled";

53 options.set<VariableName>("fixed_values") = VariableName("forces", "fixed_values");
54 options.set("fixed_values").doc() = "The name of the fixed values, i.e. the actual strain or "

77void
78MixedControlSetup::set_value(bool out, bool dout_din, bool d2out_din2)
79{
80 auto [dstrain, dstress] = make_operators(_control.tensor());
81
82 if (out)
83 {
84 // Even benign in place operations get errors
85 _stress = dstress * _fixed_values + dstrain * _mixed_state;
86 _strain = dstrain * _fixed_values + dstress * _mixed_state;
87 }
88
89 if (dout_din)
90 {
91 _stress.d(_fixed_values) = dstress;
92 _stress.d(_mixed_state) = dstrain;
93
94 _strain.d(_fixed_values) = dstrain;
95 _strain.d(_mixed_state) = dstress;
96 }
97
98 // All zero
99 (void)(d2out_din2);
100}

102std::pair<SSR4, SSR4>
103MixedControlSetup::make_operators(const SR2 & control) const
104{
105 auto strain_select = control <= _threshold;
106 auto stress_select = control > _threshold;
107
108 // This also converts these to floats
109 auto ones_stress = BatchTensor(strain_select.to(_stress.tensor().options()), control.batch_dim());
110 auto ones_strain = BatchTensor::ones_like(control) - ones_stress;
111
112 // auto dstrain = SSR4(BatchTensor(torch::diag_embed(ones_stress), batch_dim()));
113 auto dstrain = math::base_diag_embed(ones_stress);
114 auto dstress = math::base_diag_embed(ones_strain);
115
116 return {dstrain, dstress};
117}
