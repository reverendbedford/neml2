47void
48Factory::create_object(const std::string & section, const OptionSet & options)
49{
50 const std::string & name = options.name();
51 const std::string & type = options.type();
52
53 auto builder = Registry::builder(type);
54 auto object = (*builder)(options);
55 _objects[section][name].push_back(object);
56 object->setup();
57}
