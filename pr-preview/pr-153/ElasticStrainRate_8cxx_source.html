74void
75ElasticStrainRate::set_value(bool out, bool dout_din, bool d2out_din2)
76{
77 neml_assert_dbg(!d2out_din2, "Second derivative not implemented.");
78
79 if (out)
80 _e_dot = _d - _dp + math::skew_and_sym_to_sym(SR2(_e), WR2(_w));
81
82 if (dout_din)
83 {
84 const auto I = SSR4::identity_sym(options());
85 _e_dot.d(_e) = math::d_skew_and_sym_to_sym_d_sym(WR2(_w));
86 _e_dot.d(_d) = I;
87 _e_dot.d(_w) = math::d_skew_and_sym_to_sym_d_skew(SR2(_e));
88 _e_dot.d(_dp) = -I;
89 }
90}
