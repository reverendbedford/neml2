45std::tuple<bool, size_t>
46Newton::solve(NonlinearSystem & system, BatchTensor & x)
47{
48 // The initial residual for relative convergence check
49 system.residual();
50 auto nR = system.residual_norm();
51 auto nR0 = nR.clone();
52
53 // Check for initial convergence
54 if (converged(0, nR0, nR0))
55 {
56 // TODO: The final update is only necessary if we use AD
57 system.Jacobian();
58 final_update(system, x);
59 return {true, 0};
60 }
61
62 // Prepare any solver internal data before the iterative update
63 prepare(system, x);
64
65 // Continuing iterating until one of:
66 // 1. nR < atol (success)
67 // 2. nR / nR0 < rtol (success)
68 // 3. i > miters (failure)
69 for (size_t i = 1; i < miters; i++)
70 {
71 system.Jacobian();
72 update(system, x);
73 system.residual();
74 nR = system.residual_norm();
75
76 // Check for convergence
77 if (converged(i, nR, nR0))
78 {
79 // TODO: The final update is only necessary if we use AD
80 system.Jacobian();
81 final_update(system, x);
82 return {true, i};
83 }
84 }
85
86 return {false, miters};
87}

89bool
90Newton::converged(size_t itr, const torch::Tensor & nR, const torch::Tensor & nR0) const
91{
92 // LCOV_EXCL_START
93 if (verbose)
94 std::cout << "ITERATION " << std::setw(3) << itr << ", |R| = " << std::scientific
95 << torch::max(nR).item<Real>() << ", |R0| = " << std::scientific
96 << torch::max(nR0).item<Real>() << std::endl;
97 // LCOV_EXCL_STOP
98
99 return torch::all(torch::logical_or(nR < atol, nR / nR0 < rtol)).item<bool>();
100}

102void
103Newton::update(NonlinearSystem & system, BatchTensor & x)
104{
105 auto dx = solve_direction(system);
106
107 x.variable_data() += system.scale_direction(dx);
108 system.set_solution(x);
109}

111void
112Newton::final_update(NonlinearSystem & system, BatchTensor & x)
113{
114 auto dx = solve_direction(system);
115 x += system.scale_direction(dx);
116}

118BatchTensor
119Newton::solve_direction(const NonlinearSystem & system)
120{
121 // Special case when this is a scalar system
122 if (system.residual_view().base_dim() == 0)
123 return -system.residual_view() / system.Jacobian_view();
124
125 return -math::linalg::solve(system.Jacobian_view(), system.residual_view());
126}
