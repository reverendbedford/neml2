43template <typename Node, typename ItemType>
44class DependencyResolver
45{
46public:
47 struct Item
48 {
49 Item(Node * const node, const ItemType & item)
50 : parent(node),
51 value(item)
52 {
53 }
54
55 Node * const parent;
56 const ItemType value;
57
58 bool operator==(const Item & other) const
59 {
60 return parent == other.parent && value == other.value;
61 }
62
63 bool operator!=(const Item & other) const
64 {
65 return parent != other.parent || value != other.value;
66 }
67
68 bool operator<(const Item & other) const
69 {
70 return parent != other.parent ? (parent < other.parent) : (value < other.value);
71 }
72 };

162template <typename Node, typename ItemType>
163void
164DependencyResolver<Node, ItemType>::add_node(DependencyDefinition<ItemType> * def)
165{
166 auto node = dynamic_cast<Node *>(def);
167 _nodes.emplace(node);
168
169 for (const auto & item : node->consumed_items())
170 _consumed_items.emplace(node, item);
171
172 for (const auto & item : node->provided_items())
173 _provided_items.emplace(node, item);
174}

192template <typename Node, typename ItemType>
193void
194DependencyResolver<Node, ItemType>::build_graph()
195{
196 // Clear the previous graph
197 _item_provider_graph.clear();
198 _item_consumer_graph.clear();
199 _node_provider_graph.clear();
200 _node_consumer_graph.clear();
201 _start_nodes.clear();
202 _end_nodes.clear();
203 _in_items.clear();
204 _out_items.clear();
205
206 // Build the adjacency matrix for item providers and node providers
207 for (const auto & itemi : _consumed_items)
208 {
209 std::vector<Item> providers;
210
211 for (const auto & itemj : _provided_items)
212 {
213 // Match consumer with provider
214 if (itemi.value != itemj.value)
215 continue;
216
217 // No self dependency
218 if (itemi.parent == itemj.parent)
219 continue;
220
221 // Enforce priority
222 if (_priority[itemi.parent] > _priority[itemj.parent])
223 continue;
224
225 providers.push_back(itemj);
226 }
227
228 // If the user asks for unique providers, we should error if multiple providers have been
229 // found. Otherwise, just put the first provider into the graph.
230 if (!providers.empty())
231 {
232 if (_unique_item_provider)
233 neml_assert(
234 providers.size() == 1, "Multiple providers have been found for item ", itemi.value);
235 _item_provider_graph[itemi].insert(providers[0]);
236 if (itemi.parent)
237 _node_provider_graph[itemi.parent].insert(providers[0].parent);
238 }
239 }
240
241 // Build the adjacency matrix for item consumers
242 for (const auto & itemi : _provided_items)
243 {
244 std::vector<Item> consumers;
245
246 for (const auto & itemj : _consumed_items)
247 {
248 // Skip additional outbound item
249 if (!itemj.parent)
250 continue;
251
252 // Match provider with consumer
253 if (itemi.value != itemj.value)
254 continue;
255
256 // No self dependency
257 if (itemi.parent == itemj.parent)
258 continue;
259
260 // Enforce priority
261 if (_priority[itemi.parent] < _priority[itemj.parent])
262 continue;
263
264 consumers.push_back(itemj);
265 }
266
267 // If the user asks for unique consumers, we should error if multiple consumers have been
268 // found. Otherwise, just put the first consumer into the graph.
269 if (!consumers.empty())
270 {
271 if (_unique_item_consumer)
272 neml_assert(
273 consumers.size() == 1, "Multiple consumers have been found for item ", itemi.value);
274 _item_consumer_graph[itemi].insert(consumers[0]);
275 _node_consumer_graph[itemi.parent].insert(consumers[0].parent);
276 }
277 }
278
279 // Find start nodes
280 for (const auto & node : _nodes)
281 if (_node_provider_graph.count(node) == 0)
282 _start_nodes.insert(node);
283
284 // Find end nodes
285 for (const auto & node : _nodes)
286 if (_node_consumer_graph.count(node) == 0)
287 _end_nodes.insert(node);
288
289 // Find inbound items
290 for (const auto & item : _consumed_items)
291 if (_item_provider_graph.count(item) == 0)
292 _in_items.insert(item);
293
294 // Find outbound items
295 for (const auto & item : _provided_items)
296 if (_item_consumer_graph.count(item) == 0)
297 _out_items.insert(item);
298
299 // Additional outbound items
300 for (const auto & item : _consumed_items)
301 if (!item.parent)
302 {
303 neml_assert(_item_provider_graph.count(item),
304 "Unable to find provider of the additional outbound item ",
305 item.value);
306 for (const auto & provider : _item_provider_graph[item])
307 {
308 _out_items.insert(provider);
309 _end_nodes.insert(provider.parent);
310 }
311 }
312}

314template <typename Node, typename ItemType>
315void
316DependencyResolver<Node, ItemType>::resolve()
317{
318 build_graph();
319
320 _status.clear();
321 _resolution.clear();
322 for (const auto & node : _end_nodes)
323 if (!_status[node])
324 resolve(node);
325
326 // Make sure each node appears in the resolution once and only once
327 for (const auto & node : _nodes)
328 {
329 auto count = std::count(_resolution.begin(), _resolution.end(), node);
330 neml_assert(count > 0,
331 "Each node must appear in the dependency resolution. Node ",
332 node->name(),
333 " is missing. This is an internal error -- consider filing a bug report.");
334 neml_assert(count == 1,
335 "Each node must appear in the dependency resolution once and only once. Node ",
336 node->name(),
337 " appeared ",
338 count,
339 " times. This indicates cyclic dependency.");
340 }
341}

343template <typename Node, typename ItemType>
344void
345DependencyResolver<Node, ItemType>::resolve(Node * node)
346{
347 // Mark the current node as visiting (so that we know there is circular dependency when a
348 // "visiting" node is visited again).
349 _status[node] += 1;
350
351 // Recurse for all the dependent nodes
352 if (_node_provider_graph.count(node))
353 for (const auto & dep : _node_provider_graph[node])
354 {
355 // The dependent node must either be "not visited" or "visited".
356 // If the dependent node is "being visited", there must be cyclic dependency.
357 neml_assert(_status[dep] != 1,
358 "While resolving dependency, two nodes '",
359 node->name(),
360 "' and '",
361 dep->name(),
362 "' have (possibly indirect) cyclic dependency. The cyclic dependency can be "
363 "resolved by explicitly setting the node priorities.");
364
365 if (!_status[dep])
366 resolve(dep);
367 }
368
369 // At this point, all the dependent nodes must have been pushed into the resolution. It is
370 // therefore safe to push the current node into the resolution.
371 _resolution.push_back(node);
372
373 // Finished visiting this node
374 _status[node] += 1;
375}
