51void
52NewtonWithLineSearch::update(NonlinearSystem & system, BatchTensor & x)
53{
54 auto dx = solve_direction(system);
55
56 linesearch(system, x, dx);
57 x.variable_data() += system.scale_direction(_alpha * dx);
58 system.set_solution(x);
59}

61void
62NewtonWithLineSearch::linesearch(NonlinearSystem & system,
63 const BatchTensor & x,
64 const BatchTensor & dx)
65{
66 _alpha = Scalar::ones(x.batch_sizes(), x.options());
67
68 const auto & R = system.residual_view();
69 auto R0 = R.clone();
70 auto nR02 = math::bvv(R0, R0);
71
72 for (size_t i = 1; i < _linesearch_miter; i++)
73 {
74 system.set_solution(x + system.scale_direction(_alpha * dx));
75 system.residual();
76 auto nR2 = math::bvv(R, R);
77 auto crit = nR02 + 2.0 * _linesearch_c * _alpha * math::bvv(R0, dx);
78 if (verbose)
79 std::cout << " LS ITERATION " << std::setw(3) << i << ", alpha = " << std::scientific
80 << torch::min(_alpha).item<Real>() << ", |R| = " << std::scientific
81 << torch::max(torch::sqrt(nR2)).item<Real>() << ", |Rc| = " << std::scientific
82 << torch::min(torch::sqrt(crit)).item<Real>() << std::endl;
83
84 auto stop = torch::logical_or(nR2 <= crit, nR2 <= std::pow(atol, 2));
85
86 if (torch::all(stop).item<bool>())
87 break;
88
89 _alpha.batch_index_put({torch::logical_not(stop)},
90 _alpha.batch_index({torch::logical_not(stop)}) / _linesearch_sigma);
91 }
92}
