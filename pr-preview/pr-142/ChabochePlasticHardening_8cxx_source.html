57void
58ChabochePlasticHardening::set_value(bool out, bool dout_din, bool d2out_din2)
59{
60 neml_assert_dbg(!d2out_din2, "Chaboche model doesn't implement second derivatives.");
61
62 // The effective stress
63 auto s = SR2(_X).norm(NEML2_EPS);
64 // The part that's proportional to the plastic strain rate
65 auto g_term = 2.0 / 3.0 * _C * _NM - _g * _X;
66
67 if (out)
68 {
69 // The static recovery term
70 auto s_term = -_A * math::pow(s, _a - 1) * _X;
71 _X_dot = g_term * _gamma_dot + s_term;
72 }
73
74 if (dout_din)
75 {
76 auto I = SR2::identity_map(options());
77
78 _X_dot.d(_gamma_dot) = g_term;
79 _X_dot.d(_NM) = 2.0 / 3.0 * _C * _gamma_dot * I;
80 _X_dot.d(_X) = -_g * _gamma_dot * I -
81 _A * math::pow(s, _a - 3) * ((_a - 1) * SR2(_X).outer(SR2(_X)) + s * s * I);
82 }
83}
