50void
51PowerLawSlipRule::set_value(bool out, bool dout_din, bool d2out_din2)
52{
53 neml_assert_dbg(!d2out_din2, "Second derivative not implemented.");
54
55 // Grab the input
56 const auto rss = Scalar(_rss, batch_dim() + 1);
57 const auto tau = Scalar(_tau, batch_dim() + 1);
58
59 if (out)
60 _g = BatchTensor(_gamma0 * math::pow(abs(rss / tau), _n - 1.0) * rss / tau, batch_dim());
61
62 if (dout_din)
63 {
64 _g.d(_rss) = BatchTensor(
65 math::batch_diag_embed(_gamma0 * _n * math::pow(abs(rss / tau), _n - 1.0) / tau),
66 batch_dim());
67 _g.d(_tau) =
68 BatchTensor(math::batch_diag_embed(-_n * _gamma0 * rss * math::pow(abs(rss), _n - 1.0) /
69 math::pow(tau, _n + 1)),
70 batch_dim());
71 }
72}
