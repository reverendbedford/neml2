87void
88OptionSet::print(std::ostream & os) const
89{
90 OptionSet::const_iterator it = _values.begin();
91
92 os << " section: " << section() << '\n';
93 os << " doc: " << doc() << '\n';
94
95 while (it != _values.end())
96 {
97 os << " " << it->first << ":\n";
98 os << " type: " << it->second->type() << '\n';
99 os << " doc: " << it->second->doc() << '\n';
100 os << " suppressed: " << it->second->suppressed() << '\n';
101 os << " value: ";
102 it->second->print(os);
103 if (++it != _values.end())
104 os << '\n';
105 }
106}
