387template <typename T>
388bool
389OptionSet::contains(const std::string & name) const
390{
391 OptionSet::const_iterator it = _values.find(name);
392 if (it != _values.end())
393 if (dynamic_cast<const Option<T> *>(it->second.get()))
394 return true;
395 return false;
396}

398template <typename T>
399const T &
400OptionSet::get(const std::string & name) const
401{
402 neml_assert(this->contains<T>(name),
403 "ERROR: no option named \"",
404 name,
405 "\" found.\n\nKnown options:\n",
406 *this);
407
408 auto ptr = dynamic_cast<Option<T> *>(_values.at(name).get());
409 return ptr->get();
410}
411
412template <typename T>
413T &
414OptionSet::set(const std::string & name)
415{
416 if (!this->contains<T>(name))
417 _values[name] = std::make_unique<Option<T>>(name);
418 auto ptr = dynamic_cast<Option<T> *>(_values[name].get());
419 return ptr->set();
420}
