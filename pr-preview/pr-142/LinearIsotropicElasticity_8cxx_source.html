48void
49LinearIsotropicElasticity::set_value(bool out, bool dout_din, bool /*d2out_din2*/)
50{
51 // We need to work with the bulk modulus K and the shear modulus G so that the expression for
52 // stiffness and compliance can be unified:
53 const auto K = _E / 3 / (1 - 2 * _nu);
54 const auto G = _E / 2 / (1 + _nu);
55 const auto vf = _compliance ? 1 / (3 * K) : 3 * K;
56 const auto df = _compliance ? 1 / (2 * G) : 2 * G;
57
58 if (out)
59 _to = vf * SR2(_from).vol() + df * SR2(_from).dev();
60
61 if (dout_din)
62 {
63 const auto I = SSR4::identity_vol(options());
64 const auto J = SSR4::identity_dev(options());
65 _to.d(_from) = vf * I + df * J;
66 }
67}
