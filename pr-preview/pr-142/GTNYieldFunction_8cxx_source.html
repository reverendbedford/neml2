33{

35 options.set<CrossRef<Scalar>>("yield_stress");
36 options.set<CrossRef<Scalar>>("q1");

63void
64GTNYieldFunction::set_value(bool out, bool dout_din, bool d2out_din2)
65{
66 // Flow stress (depending on whether isotropic hardening is provided)
67 const auto sf = _h ? _s0 + (*_h) : _s0;
68
69 if (out)
70 _f = math::pow(_se / sf, 2.0) + 2 * _q1 * _phi * math::cosh(_q2 / 2.0 * _sp / sf) -
71 (1.0 + _q3 * math::pow(Scalar(_phi), 2.0));
72
73 if (dout_din)
74 {
75 _f.d(_se) = 2.0 * _se / math::pow(sf, 2.0);
76 _f.d(_sp) = _q1 * _phi * _q2 / sf * math::sinh(_q2 / 2.0 * _sp / sf);
77 _f.d(_phi) = 2.0 * _q1 * math::cosh(_q2 / 2.0 * _sp / sf) - 2.0 * _q3 * _phi;
78 if (_h)
79 _f.d(*_h) = -2 * math::pow(Scalar(_se), 2.0) / math::pow(sf, 3.0) -
80 _q1 * _phi * _q2 * _sp / math::pow(sf, 2.0) * math::sinh(_q2 / 2.0 * _sp / sf);
81
82 // Handle the case of nonlinear parameters
83 if (const auto sy = nl_param("sy"))
84 _f.d(*sy) = -2 * math::pow(Scalar(_se), 2.0) / math::pow(sf, 3.0) -
85 _q1 * _phi * _q2 * _sp / math::pow(sf, 2.0) * math::sinh(_q2 / 2.0 * _sp / sf);
86
87 if (const auto q1 = nl_param("q1"))
88 _f.d(*q1) = 2.0 * _phi * math::cosh(_q2 / 2.0 * _sp / sf);
89
90 if (const auto q2 = nl_param("q2"))
91 _f.d(*q2) = _q1 * _phi * _sp / sf * math::sinh(_q2 / 2.0 * _sp / sf);
92
93 if (const auto q3 = nl_param("q3"))
94 _f.d(*q3) = -math::pow(Scalar(_phi), 2.0);
95 }
96
97 if (d2out_din2)
98 {
99 const auto sy = nl_param("sy");
100 const auto q1 = nl_param("q1");
101 const auto q2 = nl_param("q2");
102 const auto q3 = nl_param("q3");
103
105 //
106 // The GTN yield function can be expressed as
107 //
108 // f(se, sp, phi, h; sy, q1, q2, q3)
109 //
110 // - Arguments before the semicolon are variables
111 // - Arguments after the semicolon are (nonlinear) parameters
112 // - Derivatives w.r.t. the first three arguments se, sp, and phi are mandatory
113 // - Derivatives w.r.t. the rest of the arguments are optional
114 //
116
118 //
119 // The second derivative is nothing but a big matrix. We will fill out the matrix row by row, in
120 // the order of the arguments listed above.
121 //
122 // Rows will separated by big fences like this.
123 //
125
127 //
128 // f(se, sp, phi, h; sy, q1, q2, q3)
129 //
130 // se: Flow invariant
131 //
133 _f.d(_se, _se) = 2.0 / math::pow(sf, 2.0);
134
135 if (_h)
136 _f.d(_se, *_h) = -4.0 * _se / math::pow(sf, 3.0);
137
138 if (sy)
139 _f.d(_se, *sy) = -4.0 * _se / math::pow(sf, 3.0);
140
142 //
143 // f(se, sp, phi, h; sy, q1, q2, q3)
144 //
145 // sp: Poro invariant
146 //
148 _f.d(_sp, _sp) = _phi * _q1 * math::pow(_q2, 2.0) / (2.0 * math::pow(sf, 2.0)) *
149 math::cosh(_q2 / 2.0 * _sp / sf);
150
151 _f.d(_sp, _phi) = _q1 * _q2 * math::sinh(_q2 / 2.0 * _sp / sf) / sf;
152
153 if (_h)
154 _f.d(_sp, *_h) = -_phi * _q1 * _q2 *
155 (_q2 * _sp * math::cosh(_q2 / 2.0 * _sp / sf) +
156 2 * sf * math::sinh(_q2 / 2.0 * _sp / sf)) /
157 (2 * math::pow(sf, 3.0));
158 if (sy)
159 _f.d(_sp, *sy) = -_phi * _q1 * _q2 *
160 (_q2 * _sp * math::cosh(_q2 / 2.0 * _sp / sf) +
161 2 * sf * math::sinh(_q2 / 2.0 * _sp / sf)) /
162 (2 * math::pow(sf, 3.0));
163
164 if (q1)
165 _f.d(_sp, *q1) = _phi * _q2 * math::sinh(_q2 / 2.0 * _sp / sf) / sf;
166
167 if (q2)
168 _f.d(_sp, *q2) = _phi * _q1 *
169 (_q2 * _sp * math::cosh(_q2 / 2.0 * _sp / sf) +
170 2.0 * sf * math::sinh(_q2 / 2.0 * _sp / sf)) /
171 (2.0 * math::pow(sf, 2.0));
172
174 //
175 // f(se, sp, phi, h; sy, q1, q2, q3)
176 //
177 // phi: Void fraction
178 //
180 _f.d(_phi, _sp) = _q1 * _q2 * math::sinh(_q2 / 2.0 * _sp / sf) / sf;
181
182 _f.d(_phi, _phi) = -2.0 * _q3;
183
184 if (_h)
185 _f.d(_phi, *_h) = -_q1 * _q2 * _sp * math::sinh(_q2 / 2.0 * _sp / sf) / math::pow(sf, 2.0);
186
187 if (sy)
188 _f.d(_phi, *sy) = -_q1 * _q2 * _sp * math::sinh(_q2 / 2.0 * _sp / sf) / math::pow(sf, 2.0);
189
190 if (q1)
191 _f.d(_phi, *q1) = 2 * math::cosh(_q2 / 2.0 * _sp / sf);
192
193 if (q2)
194 _f.d(_phi, *q2) = _q1 * _sp * math::sinh(_q2 / 2.0 * _sp / sf) / sf;
195
196 if (q3)
197 _f.d(_phi, *q3) = -2.0 * _phi;
198
200 //
201 // f(se, sp, phi, h; sy, q1, q2, q3)
202 //
203 // h: (Optional) isotropic hardening
204 //
206 if (_h)
207 {
208 _f.d(*_h, _se) = -4.0 * _se / math::pow(sf, 3.0);
209
210 _f.d(*_h, _sp) = -_phi * _q1 * _q2 *
211 (_q2 * _sp * math::cosh(_q2 / 2.0 * _sp / sf) +
212 2.0 * sf * math::sinh(_q2 / 2.0 * _sp / sf)) /
213 (2.0 * math::pow(sf, 3.0));
214
215 _f.d(*_h, _phi) = -_q1 * _q2 * _sp * math::sinh(_q2 / 2.0 * _sp / sf) / math::pow(sf, 2.0);
216
217 _f.d(*_h, *_h) =
218 (12 * math::pow(Scalar(_se), 2.0) + _phi * _q1 * _q2 * _sp *
219 (_q2 * _sp * math::cosh(_q2 / 2.0 * _sp / sf) +
220 4.0 * sf * math::sinh(_q2 / 2.0 * _sp / sf))) /
221 (2 * math::pow(sf, 4.0));
222
223 if (sy)
224 _f.d(*_h, *sy) =
225 (12 * math::pow(Scalar(_se), 2.0) + _phi * _q1 * _q2 * _sp *
226 (_q2 * _sp * math::cosh(_q2 / 2.0 * _sp / sf) +
227 4.0 * sf * math::sinh(_q2 / 2.0 * _sp / sf))) /
228 (2 * math::pow(sf, 4.0));
229
230 if (q1)
231 _f.d(*_h, *q1) = -_phi * _q2 * _sp * math::sinh(_q2 / 2.0 * _sp / sf) / math::pow(sf, 2.0);
232
233 if (q2)
234 _f.d(*_h, *q2) = -_phi * _q1 * _sp *
235 (_q2 * _sp * math::cosh(_q2 / 2.0 * _sp / sf) +
236 2.0 * sf * math::sinh(_q2 / 2.0 * _sp / sf)) /
237 (2 * math::pow(sf, 3.0));
238 }
239
241 //
242 // f(se, sp, phi, h; sy, q1, q2, q3)
243 //
244 // sy: (Optionally nonlinear) yield stress
245 //
247 if (sy)
248 {
249 _f.d(*sy, _se) = -4.0 * _se / math::pow(sf, 3.0);
250
251 _f.d(*sy, _phi) = -_q1 * _q2 * _sp * math::sinh(_q2 / 2.0 * _sp / sf) / math::pow(sf, 2.0);
252
253 if (_h)
254 _f.d(*sy, *_h) =
255 (12 * math::pow(Scalar(_se), 2.0) + _phi * _q1 * _q2 * _sp *
256 (_q2 * _sp * math::cosh(_q2 / 2.0 * _sp / sf) +
257 4.0 * sf * math::sinh(_q2 / 2.0 * _sp / sf))) /
258 (2 * math::pow(sf, 4.0));
259
260 _f.d(*sy, *sy) =
261 (12 * math::pow(Scalar(_se), 2.0) + _phi * _q1 * _q2 * _sp *
262 (_q2 * _sp * math::cosh(_q2 / 2.0 * _sp / sf) +
263 4.0 * sf * math::sinh(_q2 / 2.0 * _sp / sf))) /
264 (2 * math::pow(sf, 4.0));
265
266 if (q1)
267 _f.d(*sy, *q1) = -_phi * _q2 * _sp * math::sinh(_q2 / 2.0 * _sp / sf) / math::pow(sf, 2.0);
268
269 if (q2)
270 _f.d(*sy, *q2) = -_phi * _q1 * _sp *
271 (_q2 * _sp * math::cosh(_q2 / 2.0 * _sp / sf) +
272 2.0 * sf * math::sinh(_q2 / 2.0 * _sp / sf)) /
273 (2 * math::pow(sf, 3.0));
274 }
275
277 //
278 // f(se, sp, phi, h; sy, q1, q2, q3)
279 //
280 // q1: (Optionally nonlinear) GTN parameter q1
281 //
283 if (q1)
284 {
285 _f.d(*q1, _sp) = _phi * _q2 * math::sinh(_q2 / 2.0 * _sp / sf) / sf;
286
287 _f.d(*q1, _phi) = 2.0 * math::cosh(_q2 / 2.0 * _sp / sf);
288
289 if (_h)
290 _f.d(*q1, *_h) = -_phi * _q2 * _sp * math::sinh(_q2 / 2.0 * _sp / sf) / math::pow(sf, 2.0);
291
292 if (sy)
293 _f.d(*q1, *sy) = -_phi * _q2 * _sp * math::sinh(_q2 / 2.0 * _sp / sf) / math::pow(sf, 2.0);
294
295 if (q2)
296 _f.d(*q1, *q2) = _phi * _sp * math::sinh(_q2 / 2.0 * _sp / sf) / sf;
297 }
298
300 //
301 // f(se, sp, phi, h; sy, q1, q2, q3)
302 //
303 // q2: (Optionally nonlinear) GTN parameter q2
304 //
306 if (q2)
307 {
308 _f.d(*q2, _sp) = _phi * _q1 *
309 (_q2 * _sp * math::cosh(_q2 / 2.0 * _sp / sf) +
310 2 * sf * math::sinh(_q2 / 2.0 * _sp / sf)) /
311 (2 * math::pow(sf, 2.0));
312
313 _f.d(*q2, _phi) = _q1 * _sp * math::sinh(_q2 / 2.0 * _sp / sf) / sf;
314
315 if (_h)
316 _f.d(*q2, *_h) = -_phi * _q1 * _sp *
317 (_q2 * _sp * math::cosh(_q2 / 2.0 * _sp / sf) +
318 2 * sf * math::sinh(_q2 / 2.0 * _sp / sf)) /
319 (2 * math::pow(sf, 3.0));
320
321 if (sy)
322 _f.d(*q2, *sy) = -_phi * _q1 * _sp *
323 (_q2 * _sp * math::cosh(_q2 / 2.0 * _sp / sf) +
324 2 * sf * math::sinh(_q2 / 2.0 * _sp / sf)) /
325 (2 * math::pow(sf, 3.0));
326
327 if (q1)
328 _f.d(*q2, *q1) = _phi * _sp * math::sinh(_q2 / 2.0 * _sp / sf) / sf;
329
330 _f.d(*q2, *q2) = _phi * _q1 * math::pow(Scalar(_sp), 2.0) * math::cosh(_q2 / 2.0 * _sp / sf) /
331 (2 * math::pow(sf, 2.0));
332 }
333
335 //
336 // f(se, sp, phi, h; sy, q1, q2, q3)
337 //
338 // q3: (Optionally nonlinear) GTN parameter q3
339 //
341 if (q3)
342 {
343 _f.d(*q3, _phi) = -2.0 * _phi;
344 }
345 }
