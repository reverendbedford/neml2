60void
61WR2ImplicitExponentialTimeIntegration::set_value(bool out, bool dout_din, bool d2out_din2)
62{
63 neml_assert_dbg(!d2out_din2, "Second derivative not implemented.");
64
65 const auto dt = _t - _tn;
66 const auto inc = (_s_dot * dt).exp();
67
68 if (out)
69 _r = _s - Rot(_sn).rotate(inc);
70
71 if (dout_din)
72 {
73 const auto de = (_s_dot * dt).dexp();
74 _r.d(_s) = R2::identity(options());
75 _r.d(_s_dot) = -Rot(_sn).drotate(inc) * de * dt;
76 if (Model::stage == Model::Stage::UPDATING)
77 {
78 _r.d(_sn) = -Rot(_sn).drotate_self(inc);
79 _r.d(_t) = -Rot(_sn).drotate(inc) * de * Vec(_s_dot.value());
80 _r.d(_tn) = Rot(_sn).drotate(inc) * de * Vec(_s_dot.value());
81 }
82 }
