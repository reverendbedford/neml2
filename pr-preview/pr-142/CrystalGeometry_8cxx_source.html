154Vec
155CrystalGeometry::make_reciprocal_lattice(const Vec & lattice_vectors)
156{
157 auto a1 = lattice_vectors.batch_index({0});
158 auto a2 = lattice_vectors.batch_index({1});
159 auto a3 = lattice_vectors.batch_index({2});
160
161 Vec rl = Vec(torch::stack({a2.cross(a3) / a1.dot(a2.cross(a3)),
162 a3.cross(a1) / a2.dot(a3.cross(a1)),
163 a1.cross(a2) / a3.dot(a1.cross(a2))}));
164
165 return rl;
166}

169CrystalGeometry::miller_to_cartesian(const Vec & A, const MillerIndex & d)
170{
171 // Take advantage that a collection of 3 vectors is a R2
172 return R2(torch::Tensor(A)) * d.reduce().to_vec();

175std::tuple<Vec, Vec, Scalar, std::vector<TorchSize>>
176CrystalGeometry::setup_schmid_tensors(const Vec & A,
177 const R2 & cls,
178 const MillerIndex & slip_directions,
179 const MillerIndex & slip_planes)
180{
181 // We need the reciprocol lattice
182 Vec B = make_reciprocal_lattice(A);
183
184 // List of slip directions and planes needs to be consistent
185 if (slip_directions.batch_sizes() != slip_planes.batch_sizes())
186 neml_assert("Input slip directions and planes must have the same batch sizes");
187
188 auto bshape = slip_planes.batch_sizes();
189 auto nbatch = slip_planes.batch_dim();
190
191 // Loop over each slip system
192 std::vector<torch::Tensor> cartesian_slip_directions;
193 std::vector<torch::Tensor> cartesian_slip_planes;
194 std::vector<torch::Tensor> burgers_vectors;
195 std::vector<TorchSize> offsets = {0};
196
197 for (TorchSize i = 0; i < bshape[nbatch - 1]; i++)
198 {
199 // Get the cartesian slip plane and direction
200 auto cmd = slip_directions.batch_index({torch::indexing::Ellipsis, i});
201 auto cmp = slip_planes.batch_index({torch::indexing::Ellipsis, i});
202
203 // Get the families of symmetry-equivalent planes and directions
204 auto direction_options = unique_bidirectional(cls, miller_to_cartesian(A, cmd));
205 auto plane_options = unique_bidirectional(cls, miller_to_cartesian(B, cmp));
206
207 // Accept the ones that are perpendicular
208 // We could do this in a vectorized manner, but I don't think it's worth it as
209 // this code only runs once
210 TorchSize last = offsets.back();
211 for (TorchSize j = 0; j < direction_options.batch_sizes()[direction_options.batch_dim() - 1];
212 j++)
213 {
214 auto di = direction_options.batch_index({torch::indexing::Ellipsis, j});
215 auto dps = plane_options.dot(di);
216 auto inds =
217 torch::where(torch::isclose(torch::abs(dps), torch::tensor(0.0, dps.dtype()))).front();
218 // We could very easily vectorize this loop, but again whatever
219 for (TorchSize kk = 0; kk < inds.sizes()[0]; kk++)
220 {
221 TorchSize k = inds.index({kk}).item<TorchSize>();
222 auto pi = plane_options.batch_index({torch::indexing::Ellipsis, k});
223 cartesian_slip_directions.push_back(di / di.norm());
224 cartesian_slip_planes.push_back(pi / pi.norm());
225 burgers_vectors.push_back(di.norm());
226 last += 1;
227 }
228 }
229 offsets.push_back(last);
230 }
231
232 return std::make_tuple(Vec(torch::stack(cartesian_slip_directions)),
233 Vec(torch::stack(cartesian_slip_planes)),
234 Scalar(torch::stack(burgers_vectors)),
235 offsets);
236}
