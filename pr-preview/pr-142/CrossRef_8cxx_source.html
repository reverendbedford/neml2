33template <>
34CrossRef<torch::Tensor>::operator torch::Tensor() const
35{
36 try
37 {
38 // If it is just a number, we can still create a tensor out of it
39 return torch::tensor(utils::parse<Real>(_raw_str), default_tensor_options());
40 }
41 catch (const ParserException & e)
42 {
43 // Conversion to a number failed, so it might be the name of another tensor
44 return Factory::get_object<torch::Tensor>("Tensors", _raw_str);
45 }
46}
