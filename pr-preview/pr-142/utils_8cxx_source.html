38TorchShape
39pad_prepend(TorchShapeRef s, TorchSize dim, TorchSize pad)
40{
41 auto s2 = s.vec();
42 s2.insert(s2.begin(), dim - s.size(), pad);
43 return s2;
44}

46TorchShape
47pad_append(TorchShapeRef s, TorchSize dim, TorchSize pad)
48{
49 auto s2 = s.vec();
50 s2.insert(s2.end(), dim - s.size(), pad);
51 return s2;
52}
