41ImplicitUpdate::ImplicitUpdate(const OptionSet & options)
42 : Model(options),
43 _model(register_model<Model>(options.get<std::string>("implicit_model"),
44 /*extra_deriv_order=*/requires_grad() ? 0 : 1,
45 /*nonlinear=*/true)),
46 _solver(Factory::get_object<NonlinearSolver>("Solvers", options.get<std::string>("solver")))
47{
48 // Make sure the nonlinear system is square
49 neml_assert(_model.input_axis().has_subaxis("state"),
50 "The implicit model's input should have a state subaxis. The input axis is\n",
51 _model.input_axis());
52 neml_assert(_model.output_axis().has_subaxis("residual"),
53 "The implicit model's output should have a residual subaxis. The output axis is\n",
54 _model.output_axis());
55 neml_assert(_model.input_axis().subaxis("state") == _model.output_axis().subaxis("residual"),
56 "The implicit model should have conformal trial state and residual. The input state "
57 "subaxis is\n",
58 _model.input_axis().subaxis("state"),
59 "\nThe output residual subaxis is\n",
60 _model.output_axis().subaxis("residual"));
61
62 // Take care of dependency registration:
63 // 1. Input variables of the "implicit_model" should be *consumed* by *this* model. This has
64 // already been taken care of by the `register_model` call.
65 // 2. Output variables of the "implicit_model" on the "residual" subaxis should be *provided* by
66 // *this* model.
67 for (auto var : _model.output_axis().subaxis("residual").variable_accessors(/*recursive=*/true))
68 declare_output_variable(_model.output_axis().subaxis("residual").storage_size(var),
69 var.on("state"));
70}

80void
81ImplicitUpdate::set_value(bool out, bool dout_din, bool d2out_din2)
82{
83 neml_assert_dbg(!d2out_din2, "This model does not define the second derivatives.");
84 neml_assert_dbg(
85 !dout_din || out,
86 "ImplicitUpdate: requires the value and the first derivatives to be computed together.");
87
88 // Apply initial guess
89 LabeledVector sol0(_model.solution(), {&output_axis()});
90 sol0.fill(host<VariableStore>()->input_storage());
91
92 // The trial state is used as the initial guess
93 // Perform automatic scaling
94 _model.init_scaling(_solver.verbose);
95
96 if (out || dout_din)
97 {
98 // Solve for the next state
99 Model::stage = Model::Stage::SOLVING;
100 auto sol = _model.solution().clone();
101 auto [succeeded, iters] = _solver.solve(_model, sol);
102 neml_assert(succeeded, "Nonlinear solve failed.");
103 Model::stage = Model::Stage::UPDATING;
104
105 if (out)
106 output_storage().copy_(sol);
107
108 // Use the implicit function theorem (IFT) to calculate the other derivatives
109 if (dout_din)
110 {
111 // IFT requires dresidual/dinput evaluated at the solution:
112 _model.value_and_dvalue();
113 const auto & partials = _model.derivative_storage();
114
115 // The actual IFT:
116 LabeledMatrix J = partials.slice(1, "state");
117 auto [LU, pivot] = math::linalg::lu_factor(J);
118 derivative_storage()("state", "old_state")
119 .copy_(-math::linalg::lu_solve(LU, pivot, partials.slice(1, "old_state")));
120 derivative_storage()("state", "forces")
121 .copy_(-math::linalg::lu_solve(LU, pivot, partials.slice(1, "forces")));
122 derivative_storage()("state", "old_forces")
123 .copy_(-math::linalg::lu_solve(LU, pivot, partials.slice(1, "old_forces")));
124 }
125 }
126}
