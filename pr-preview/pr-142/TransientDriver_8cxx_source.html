34TransientDriver::expected_options()

58TransientDriver::TransientDriver(const OptionSet & options)
59 : Driver(options),
60 _model(Factory::get_object<Model>("Models", options.get<std::string>("model"))),
61 _device(options.get<std::string>("device")),
62 _time(options.get<CrossRef<torch::Tensor>>("times"), 2),
63 _step_count(0),
64 _time_name(options.get<VariableName>("time")),
65 _nsteps(_time.batch_sizes()[0]),
66 _nbatch(_time.batch_sizes()[1]),
67 _in(_model.input_storage()),
68 _out(_model.output_storage()),
69 _predictor(options.get<std::string>("predictor")),
70 _save_as(options.get<std::string>("save_as")),
71 _show_params(options.get<bool>("show_parameters")),
72 _show_input(options.get<bool>("show_input_axis")),
73 _show_output(options.get<bool>("show_output_axis")),
74 _result_in(LabeledVector::zeros({_nsteps, _nbatch}, {&_model.input_axis()})),
75 _result_out(LabeledVector::zeros({_nsteps, _nbatch}, {&_model.output_axis()})),
76 _ic_scalar_names(options.get<std::vector<VariableName>>("ic_scalar_names")),
77 _ic_scalar_values(options.get<std::vector<CrossRef<Scalar>>>("ic_scalar_values")),
78 _ic_rot_names(options.get<std::vector<VariableName>>("ic_rot_names")),
79 _ic_rot_values(options.get<std::vector<CrossRef<Rot>>>("ic_rot_values")),
80 _ic_sr2_names(options.get<std::vector<VariableName>>("ic_sr2_names")),
81 _ic_sr2_values(options.get<std::vector<CrossRef<SR2>>>("ic_sr2_values")),
82 _cp_elastic_scale(options.get<Real>("cp_elastic_scale"))
83{
84 _model.reinit({_nbatch}, 0, _device);
85
86 _time = _time.to(_device);
87 _result_in = _result_in.to(_device);
88 _result_out = _result_out.to(_device);
89}

129bool
130TransientDriver::solve()
131{
132 for (_step_count = 0; _step_count < _nsteps; _step_count++)
133 {
134 if (_verbose)
135 // LCOV_EXCL_START
136 std::cout << "Step " << _step_count << std::endl;
137 // LCOV_EXCL_STOP
138
139 if (_step_count > 0)
140 advance_step();
141 update_forces();
142 if (_step_count == 0)
143 {
144 store_input();
145 apply_ic();
146 }
147 else
148 {
149 apply_predictor();
150 store_input();
151 solve_step();
152 }
153 store_output();
154
155 if (_verbose)
156 // LCOV_EXCL_START
157 std::cout << std::endl;
158 // LCOV_EXCL_STOP
159 }
160
161 return true;
162}

164void
165TransientDriver::advance_step()
166{
167 if (_in.axis(0).has_subaxis("old_state") && _out.axis(0).has_subaxis("state"))
168 _in.slice("old_state").fill(_out.slice("state"));
169
170 if (_in.axis(0).has_subaxis("old_forces") && _in.axis(0).has_subaxis("forces"))
171 _in.slice("old_forces").fill(_in.slice("forces"));
172}

189void
190TransientDriver::apply_predictor()
191{
192 std::string predictor = _predictor;
193 bool cp = false;
194 if (predictor.substr(0, 3) == "CP_")
195 {
196 predictor = predictor.substr(3);
197 cp = true;
198 }
199
200 if (_in.axis(0).has_subaxis("state") && _in.axis(0).has_subaxis("old_state"))
201 {
202 if (predictor == "PREVIOUS_STATE")
203 _in.slice("state").fill(_in.slice("old_state"));
204 else if (predictor == "LINEAR_EXTRAPOLATION")
205 {
206 // Fall back to PREVIOUS_STATE predictor at the 1st time step
207 if (_step_count == 1)
208 _in.slice("state").fill(_in.slice("old_state"));
209 // Otherwise linearly extrapolate in time
210 else
211 {
212 auto t = _in.get<Scalar>(_time_name);
213 auto t_n = _result_in.get<Scalar>(_time_name).batch_index({_step_count - 1});
214 auto t_nm1 = _result_in.get<Scalar>(_time_name).batch_index({_step_count - 2});
215 auto dt = t - t_n;
216 auto dt_n = t_n - t_nm1;
217
218 auto states = _result_out.slice("state");
219 auto state_n = states.tensor().batch_index({_step_count - 1});
220 auto state_nm1 = states.tensor().batch_index({_step_count - 2});
221 LabeledVector state(state_n + (state_n - state_nm1) / dt_n * dt, states.axes());
222 _in.slice("state").fill(state);
223 }
224 }
225 else
226 throw NEMLException("Unrecognized predictor type: " + _predictor);
227 }
228
229 if (cp && (_step_count == 1))
230 {
231 SR2 D = _in.get<SR2>(std::vector<std::string>{"forces", "deformation_rate"});
232 auto t = _in.get<Scalar>(_time_name);
233 auto t_n = _result_in.get<Scalar>(_time_name).batch_index({_step_count - 1});
234 _in.set(D * (t - t_n) * _cp_elastic_scale, std::vector<std::string>{"state", "elastic_strain"});
235 }
236}

262torch::nn::ModuleDict
263TransientDriver::result() const
264{
265 auto result_in_cpu = _result_in.to(torch::kCPU);
266 auto result_out_cpu = _result_out.to(torch::kCPU);
267
268 // Dump input variables into a Module
269 auto res_in = std::make_shared<torch::nn::Module>();
270 for (auto var : result_in_cpu.axis(0).variable_accessors(/*recursive=*/true))
271 res_in->register_buffer(utils::stringify(var), result_in_cpu(var).clone());
272
273 // Dump output variables into a Module
274 auto res_out = std::make_shared<torch::nn::Module>();
275 for (auto var : result_out_cpu.axis(0).variable_accessors(/*recursive=*/true))
276 res_out->register_buffer(utils::stringify(var), result_out_cpu(var).clone());
277
278 // Combine input and output
279 torch::nn::ModuleDict res;
280 res->update({{"input", res_in}, {"output", res_out}});
281 return res;
282}
