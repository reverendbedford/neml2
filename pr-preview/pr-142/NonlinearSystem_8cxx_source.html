64void
65NonlinearSystem::init_scaling(const bool verbose)
66{
67 if (!_autoscale)
68 return;
69
70 if (_scaling_matrices_initialized)
71 return;
72
73 using namespace torch::indexing;
74
75 // First compute the Jacobian
76 assemble(false, true);
77
78 auto Jp = _Jacobian.clone();
79 _row_scaling = BatchTensor::ones_like(_solution);
80 _col_scaling = BatchTensor::ones_like(_solution);
81
82 if (verbose)
83 std::cout << "Before automatic scaling cond(J) = " << std::scientific
84 << torch::max(torch::linalg_cond(Jp)).item<Real>() << std::endl;
85
86 for (unsigned int itr = 0; itr < _autoscale_miter; itr++)
87 {
88 // check for convergence
89 auto rR = torch::max(torch::abs(1.0 - 1.0 / torch::sqrt(std::get<0>(Jp.max(-1))))).item<Real>();
90 auto rC = torch::max(torch::abs(1.0 - 1.0 / torch::sqrt(std::get<0>(Jp.max(-2))))).item<Real>();
91 if (verbose)
92 std::cout << "ITERATION " << itr << ", ROW ILLNESS = " << std::scientific << rR
93 << ", COL ILLNESS = " << std::scientific << rC << std::endl;
94 if (rR < _autoscale_tol && rC < _autoscale_tol)
95 break;
96
97 // scale rows and columns
98 for (TorchSize i = 0; i < _ndof; i++)
99 {
100 auto ar = 1.0 / torch::sqrt(torch::max(Jp.base_index({i})));
101 auto ac = 1.0 / torch::sqrt(torch::max(Jp.base_index({Slice(), i})));
102 _row_scaling.base_index({i}) *= ar;
103 _col_scaling.base_index({i}) *= ac;
104 Jp.base_index({i}) *= ar;
105 Jp.base_index({Slice(), i}) *= ac;
106 }
107 }
108
109 _scaling_matrices_initialized = true;
110
111 if (verbose)
112 std::cout << " After automatic scaling cond(J) = " << std::scientific
113 << torch::max(torch::linalg_cond(Jp)).item<Real>() << std::endl;
114}

116BatchTensor
117NonlinearSystem::scale_residual(const BatchTensor & r) const
118{
119 neml_assert_dbg(
120 _autoscale == _scaling_matrices_initialized,
121 _autoscale ? "Automatic scaling is requested but scaling matrices have not been initialized."
122 : "Automatic scaling is not requested but scaling matrices were initialized.");
123 return _row_scaling * r;
124}
125
126BatchTensor
127NonlinearSystem::scale_Jacobian(const BatchTensor & J) const
128{
129 neml_assert_dbg(
130 _autoscale == _scaling_matrices_initialized,
131 _autoscale ? "Automatic scaling is requested but scaling matrices have not been initialized."
132 : "Automatic scaling is not requested but scaling matrices were initialized.");
133 return math::bmm(math::bmm(math::base_diag_embed(_row_scaling), J),
134 math::base_diag_embed(_col_scaling));
135}

137BatchTensor
138NonlinearSystem::scale_direction(const BatchTensor & p) const
139{
140 neml_assert_dbg(
141 _autoscale == _scaling_matrices_initialized,
142 _autoscale ? "Automatic scaling is requested but scaling matrices have not been initialized."
143 : "Automatic scaling is not requested but scaling matrices were initialized.");
144 return _autoscale ? _col_scaling * p : p;
145}
