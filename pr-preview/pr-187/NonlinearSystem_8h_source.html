220template <bool scaled>
221NonlinearSystem::Res<scaled>
222NonlinearSystem::residual()
223{
224 Res<false> r;
225 assemble(&r, nullptr);
226 if constexpr (scaled)
227 return scale(r);
228 else
229 return r;
230}

232template <bool scaled>

240template <bool scaled>
241NonlinearSystem::Jac<scaled>
242NonlinearSystem::Jacobian()
243{

260template <bool scaled>
261std::tuple<NonlinearSystem::Res<scaled>, NonlinearSystem::Jac<scaled>>
262NonlinearSystem::residual_and_Jacobian()
263{
264 Res<false> r;
265 Jac<false> J;
266 assemble(&r, &J);
267 if constexpr (scaled)
268 return {scale(r), scale(J)};
269 else
270 return {r, J};
271}
