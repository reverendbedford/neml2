31void
32load_model(const std::string & path, const std::string & additional_input, ParserType ptype)
33{
34 // We are being forward looking here
35 if (ptype == ParserType::AUTO)
36 {
37 if (utils::end_with(path, ".i"))
38 ptype = ParserType::HIT;
39 else if (utils::end_with(path, ".xml"))
40 ptype = ParserType::XML;
41 else if (utils::end_with(path, ".yml"))
42 ptype = ParserType::YAML;
43 }
44
45 // but for now we only support HIT
46 if (ptype == ParserType::HIT)
47 {
48 HITParser parser;
49
50 Factory::clear();
51 Factory::load(parser.parse(path, additional_input));
52 }
53 else
54 neml_assert(false, "Unsupported parser type");
55}
