50UserBatchTensor::UserBatchTensor(const OptionSet & options)
51 : BatchTensor(BatchTensor::empty(options.get<TorchShape>("batch_shape"),
52 options.get<TorchShape>("base_shape"),
53 default_tensor_options())),
54 UserTensor(options)
55{
56 auto vals = options.get<std::vector<Real>>("values");
57 auto flat = torch::tensor(vals, default_tensor_options());
58 if (vals.size() == size_t(this->base_storage()))
59 this->index_put_({torch::indexing::Ellipsis}, flat.reshape(this->base_sizes()));
60 else if (vals.size() == size_t(utils::storage_size(this->sizes())))
61 this->index_put_({torch::indexing::Ellipsis}, flat.reshape(this->sizes()));
62 else
63 neml_assert(false,
64 "Number of values ",
65 vals.size(),
66 " must equal to either the base storage size ",
67 this->base_storage(),
68 " or the total storage size ",
69 utils::storage_size(this->sizes()));
70}
