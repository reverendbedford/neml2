64template <typename T>
65void
66parse_(T & val, const std::string & raw_str)
67{
68 std::stringstream ss(trim(raw_str));
69 ss >> val;
70 if (ss.fail() || !ss.eof())
71 throw ParserException("Failed to parse '" + raw_str + "' as a " +
72 utils::demangle(typeid(T).name()));
73}
74
75template <typename T>
76T
77parse(const std::string & raw_str)
78{
79 T val;
80 parse_(val, raw_str);
81 return val;
82}

84template <typename T>
85void
86parse_vector_(std::vector<T> & vals, const std::string & raw_str)
87{
88 auto tokens = split(raw_str, " \t\n\v\f\r");
89 vals.resize(tokens.size());
90 for (size_t i = 0; i < tokens.size(); i++)
91 parse_<T>(vals[i], tokens[i]);
92}
93
94template <typename T>
95std::vector<T>

97{
98 std::vector<T> vals;

101}

103template <typename T>
104void
105parse_vector_vector_(std::vector<std::vector<T>> & vals, const std::string & raw_str)
106{
107 auto token_vecs = split(raw_str, ";");
108 vals.resize(token_vecs.size());
109 for (size_t i = 0; i < token_vecs.size(); i++)
110 parse_vector_<T>(vals[i], token_vecs[i]);
111}
