131 template <typename T>
132 void set_ic(const std::string & name_opt, const std::string & value_opt)
133 {
134 const auto names = input_options().get<std::vector<VariableName>>(name_opt);
135 const auto vals = input_options().get<std::vector<CrossRef<T>>>(value_opt);
136 neml_assert(names.size() == vals.size(),
137 "Number of initial condition names ",
138 name_opt,
139 " and number of initial condition values ",
140 value_opt,
141 " should be the same but instead have ",
142 names.size(),
143 " and ",
144 vals.size(),
145 " respectively.");
146 for (std::size_t i = 0; i < names.size(); i++)
147 {
148 neml_assert(names[i].start_with("state"),
149 "Initial condition names should start with 'state' but instead got ",
150 names[i]);
151 _result_out[0][names[i]] = T(vals[i]).to(_device);
152 }
153 }
