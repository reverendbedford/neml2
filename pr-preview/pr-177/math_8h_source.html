234template <class T, typename = typename std::enable_if_t<std::is_base_of_v<TensorBase<T>, T>>>
235T
236batch_cat(const std::vector<T> & tensors, Size d = 0)
237{
238 neml_assert_dbg(!tensors.empty(), "batch_cat must be given at least one tensor");
239 std::vector<torch::Tensor> torch_tensors(tensors.begin(), tensors.end());
240 auto d2 = d >= 0 ? d : d - tensors.begin()->base_dim();
241 return T(torch::cat(torch_tensors, d2), tensors.begin()->batch_dim());
242}

259T
260batch_sum(const T & a, Size d = 0)
261{
262 neml_assert_dbg(a.batch_dim() > 0, "Must have a batch dimension to sum along");
263 auto d2 = d >= 0 ? d : d - a.base_dim();
