439template <typename T>
440bool
441OptionSet::Option<T>::operator==(const OptionBase & other) const
442{
443 const auto other_ptr = dynamic_cast<const Option<T> *>(&other);
444 if (!other_ptr)
445 return false;
446
447 return (_metadata == other_ptr->_metadata) && (other_ptr->get() == this->get());
448}

476template <typename T>
477bool
478OptionSet::contains(const std::string & name) const
479{
480 OptionSet::const_iterator it = _values.find(name);
481 if (it != _values.end())
482 if (dynamic_cast<const Option<T> *>(it->second.get()))
483 return true;
484 return false;
485}

487template <typename T>
488const T &
489OptionSet::get(const std::string & name) const
490{
491 neml_assert(this->contains<T>(name),
492 "ERROR: no option named \"",
493 name,
494 "\" found.\n\nKnown options:\n",
495 *this);
496
497 auto ptr = dynamic_cast<Option<T> *>(_values.at(name).get());
498 return ptr->get();
499}

501template <typename T, FType F>
502T &
503OptionSet::set(const std::string & name)
504{
505 if (!this->contains<T>(name))
506 _values[name] = std::make_unique<Option<T>>(name);
507 auto ptr = dynamic_cast<Option<T> *>(_values[name].get());
508 ptr->ftype() = F;
509 return ptr->set();
510}
