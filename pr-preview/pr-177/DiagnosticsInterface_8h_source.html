94template <typename... Args>
95void
96DiagnosticsInterface::diagnostic_assert(std::vector<Diagnosis> & diagnoses,
97 bool assertion,
98 Args &&... args) const
99{
100 if (assertion)
101 return;
102
103 std::ostringstream oss;
104 internal::stream_all(oss,
105 "In object '",
106 _object->name(),
107 "' of type ",
108 _object->type(),
109 ": ",
110 std::forward<Args>(args)...);
111 diagnoses.emplace_back(Diagnosis(oss.str()));
112}
