89template <class Derived>
90Derived
91TensorBase<Derived>::linspace(const Derived & start, const Derived & end, Size nstep, Size dim)
92{
93 neml_assert_broadcastable_dbg(start, end);
94 neml_assert_dbg(nstep > 0, "nstep must be positive.");
95
96 auto res = start.batch_unsqueeze(dim);
97
98 if (nstep > 1)
99 {
100 auto Bd = broadcast_batch_dim(start, end);
101 auto diff = (end - start).batch_unsqueeze(dim);
102
103 indexing::TensorIndices net(dim, indexing::None);
104 net.push_back(indexing::Ellipsis);
105 net.insert(net.end(), Bd - dim, indexing::None);
106 Scalar steps = torch::arange(nstep, diff.options()).index(net) / (nstep - 1);
107
108 res = res + steps * diff;
109 }
110
111 return res;
112}

270template <class Derived>
271Derived
272TensorBase<Derived>::batch_expand(const TraceableTensorShape & batch_shape) const
273{
274 // We don't want to touch the base dimensions, so put -1 for them.
275 auto net = batch_shape.concrete();
276 net.insert(net.end(), base_dim(), -1);
277
278 // Record the batch sizes in the traced graph if we are tracing
279 for (Size i = 0; i < (Size)batch_shape.size(); ++i)
280 if (const auto * const si = batch_shape[i].traceable())
281 torch::jit::tracer::ArgumentStash::stashIntArrayRefElem("size", net.size(), i, *si);
282
283 return Derived(expand(net), batch_shape);
284}

286template <class Derived>
287Derived
288TensorBase<Derived>::batch_expand(const TraceableSize & batch_size, Size dim) const
289{
290 // We don't want to touch other batch dimensions and the base dimensions, so put -1 for them.
291 auto net = std::vector<Size>(this->dim(), -1);
292 auto i = dim >= 0 ? dim : this->dim() + dim - base_dim();
293 net[i] = batch_size.concrete();
294
295 // Record the batch sizes in the traced graph if we are tracing
296 if (const auto * const s = batch_size.traceable())
297 torch::jit::tracer::ArgumentStash::stashIntArrayRefElem("size", this->dim(), i, *s);
298
299 return Derived(expand(net), batch_dim());
300}

302template <class Derived>
303neml2::Tensor
304TensorBase<Derived>::base_expand(TensorShapeRef base_shape) const
305{
306 if (base_sizes() == base_shape)
307 return *this;
308
309 // We don't want to touch the batch dimensions, so put -1 for them.
310 auto net = base_shape.vec();
311 net.insert(net.begin(), batch_dim(), -1);
312 return neml2::Tensor(expand(net), batch_sizes());
313}

315template <class Derived>
316neml2::Tensor
317TensorBase<Derived>::base_expand(Size base_size, Size dim) const
318{
319 if (this->base_size(dim) == base_size)
320 return *this;
321
322 // We don't want to touch the batch dimensions and other base dimensions, so put -1 for them.
323 auto net = std::vector<Size>(this->dim(), -1);
324 auto i = dim < 0 ? this->dim() + dim : dim + batch_dim();
325 net[i] = base_size;
326 return neml2::Tensor(expand(net), batch_sizes());
327}
