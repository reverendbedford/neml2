177 template <typename T, typename = typename std::enable_if_t<std::is_base_of_v<Model, T>>>
178 T & register_model(const std::string & name, bool nonlinear = false, bool merge_input = true)
179 {
180 neml_assert(name != this->name(),
181 "Model named '",
182 this->name(),
183 "' is trying to register itself as a sub-model. This is not allowed.");
184
185 OptionSet extra_opts;
186 extra_opts.set<NEML2Object *>("_host") = host();
187 extra_opts.set<bool>("_nonlinear_system") = nonlinear;
188
189 auto model = Factory::get_object_ptr<Model>("Models", name, extra_opts);
190
191 if (merge_input)
192 for (auto && [name, var] : model->input_variables())
193 clone_input_variable(var);
194
195 _registered_models.push_back(model.get());
196 return *(std::dynamic_pointer_cast<T>(model));
197 }
