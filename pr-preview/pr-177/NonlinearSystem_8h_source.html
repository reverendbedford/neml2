217template <bool scaled>
218NonlinearSystem::RES<scaled>
219NonlinearSystem::residual()
220{
221 RES<false> r;

223 if constexpr (scaled)

257template <bool scaled>
258std::tuple<NonlinearSystem::RES<scaled>, NonlinearSystem::JAC<scaled>>
259NonlinearSystem::residual_and_Jacobian()
260{
261 RES<false> r;
262 JAC<false> J;
263 assemble(&r, &J);
264 if constexpr (scaled)
265 return {scale(r), scale(J)};
266 else
267 return {r, J};
268}
