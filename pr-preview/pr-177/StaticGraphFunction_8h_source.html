93template <typename OutputType, typename... InputTypes>
94OutputType
95StaticGraphFunction<OutputType, InputTypes...>::operator()(const InputTypes &... args)
96{
97 auto stack = tuple_to_stack(std::make_tuple(args...));
98 _graph_function.run(stack);
99 neml_assert_dbg(stack.size() == std::tuple_size_v<OutputType>,
100 "Number of outputs on the stack does not match the expected number of outputs");
101 OutputType output;
102 details::stack_to_tuple_impl<std::tuple_size_v<OutputType> - 1>(stack, output);
103 return output;
104}

106template <typename OutputType, typename... InputTypes>
107std::shared_ptr<torch::jit::Graph>
108StaticGraphFunction<OutputType, InputTypes...>::create_graph(
109 const std::function<OutputType(InputTypes &...)> & f, const std::tuple<InputTypes...> & x) const
110{
111 auto var_name_lookup_fn = [](const torch::autograd::Variable & /*var*/) -> std::string
112 { return ""; };
113 auto f_wrap = [&f](torch::jit::Stack inputs) -> torch::jit::Stack
114 {
115 auto x = stack_to_tuple<InputTypes...>(inputs);
116 return tuple_to_stack(std::apply(f, x));
117 };
118 return std::move(std::get<0>(torch::jit::tracer::trace(tuple_to_stack(x),
119 f_wrap,
120 var_name_lookup_fn,
121 /*strict=*/true,
122 /*force_outplace=*/false))
123 ->graph);
124}
