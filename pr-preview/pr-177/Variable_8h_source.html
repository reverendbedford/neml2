350// Everything below is just for convenience: We just forward operations to the the variable values
351// so that we can do
352//
353// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
354// var4 = (var1 - var2) * var3
355// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
356//
357// instead of the (ugly?) expression below
358//
359// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
360// var4 = (var1.v - var2.v) * var3.v
361// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
362#define FWD_VARIABLE_BINARY_OP(op) \
363 template <typename T1, \
364 typename T2, \
365 typename = typename std::enable_if_t<std::is_base_of_v<VariableBase, T1> || \
366 std::is_base_of_v<VariableBase, T2>>> \
367 auto op(const T1 & a, const T2 & b) \
368 { \
369 if constexpr (std::is_base_of_v<VariableBase, T1> && std::is_base_of_v<VariableBase, T2>) \
370 return op(a.value(), b.value()); \
371 \
372 if constexpr (std::is_base_of_v<VariableBase, T1> && !std::is_base_of_v<VariableBase, T2>) \
373 return op(a.value(), b); \
374 \
375 if constexpr (!std::is_base_of_v<VariableBase, T1> && std::is_base_of_v<VariableBase, T2>) \
376 return op(a, b.value()); \
377 } \
378 static_assert(true)
379FWD_VARIABLE_BINARY_OP(operator+);
380FWD_VARIABLE_BINARY_OP(operator-);
381FWD_VARIABLE_BINARY_OP(operator*);
382FWD_VARIABLE_BINARY_OP(operator/);
