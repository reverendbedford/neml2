334// Everything below is just for convenience: We just forward operations to the the variable values
335// so that we can do
336//
337// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
338// var4 = (var1 - var2) * var3
339// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
340//
341// instead of the (ugly?) expression below
342//
343// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
344// var4 = (var1.v - var2.v) * var3.v
345// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
346#define FWD_VARIABLE_BINARY_OP(op) \
347 template <typename T1, \
348 typename T2, \
349 typename = typename std::enable_if_t<std::is_base_of_v<VariableBase, T1> || \
350 std::is_base_of_v<VariableBase, T2>>> \
351 auto op(const T1 & a, const T2 & b) \
352 { \
353 if constexpr (std::is_base_of_v<VariableBase, T1> && std::is_base_of_v<VariableBase, T2>) \
354 return op(a.value(), b.value()); \
355 \
356 if constexpr (std::is_base_of_v<VariableBase, T1> && !std::is_base_of_v<VariableBase, T2>) \
357 return op(a.value(), b); \
358 \
359 if constexpr (!std::is_base_of_v<VariableBase, T1> && std::is_base_of_v<VariableBase, T2>) \
360 return op(a, b.value()); \
361 } \
362 static_assert(true)
363FWD_VARIABLE_BINARY_OP(operator+);
364FWD_VARIABLE_BINARY_OP(operator-);
365FWD_VARIABLE_BINARY_OP(operator*);
366FWD_VARIABLE_BINARY_OP(operator/);
