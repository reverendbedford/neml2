73 template <typename Container,
74 typename = typename std::enable_if_t<
75 !std::is_convertible_v<Container, std::string> &&

79 std::is_convertible_v<typename std::iterator_traits<

81 std::string>>>
82 LabeledAxisAccessor(const Container & c)
83 {
84 _item_names.append(c.begin(), c.end());
85 for (const auto & name : _item_names)
86 validate_item_name(name);
87 }
