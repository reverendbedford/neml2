56template <std::size_t N>
57class ElasticityConverter
58{
59public:
60 using InputType = std::array<Scalar, N>;
61 using InputPtrType = std::array<const Scalar *, N>;
62 using DerivativeFlagType = std::array<bool, N>;
63 using DerivativeType = std::array<Scalar, N>;
64 using ConversionType = std::pair<Scalar, DerivativeType>;
65 using ResultType = std::array<ConversionType, N>;
66
67 using ConverterKey = std::array<LameParameter, N>;
68 using ConverterType = ConversionType (*)(const InputType &, const DerivativeFlagType &);
69 using ConversionTableType = std::map<ConverterKey, std::array<ConverterType, N>>;
70
71 ElasticityConverter(const ConversionTableType & table,
72 const ConverterKey & parameterization,
73 const DerivativeFlagType & deriv_requested)
74 : _deriv_requested(deriv_requested)
75 {
76 assert_ascending(parameterization);
77 neml_assert(table.count(parameterization),
78 "Parameterization not found in the conversion table. This typically means that the "
79 "given combination of Lame parameters is not yet supported.");
80 _converters = table.at(parameterization);
81 }
82
84 ResultType convert(const InputType & input) const
85 {
86 ResultType ret{};
87 for (std::size_t i = 0; i < N; ++i)
88 ret[i] = _converters[i](input, _deriv_requested);
89 return ret;
90 }
91
93 ResultType convert(const InputPtrType & input) const
94 {
95 InputType input_values{};
96 for (std::size_t i = 0; i < N; ++i)
97 input_values[i] = *input[i];
98 return convert(input_values);
99 }
100
101private:
103 void assert_ascending(const ConverterKey & ps) const
104 {
105 for (std::size_t i = 1; i < N; ++i)
106 neml_assert(
107 static_cast<std::uint8_t>(ps[i]) > static_cast<std::uint8_t>(ps[i - 1]),
108 "Internal error: ElasticityConverters only accept Lame parameters sorted in the "
109 "following order: LAME_FIRST_CONSTANT, BULK_MODULUS, SHEAR_MODULUS, YOUNGS_MODULUS, "
110 "POISSONS_RATIO, P_WAVE_MODULUS.");
111 }
112
113 // Default initialize the flags (false)
114 const DerivativeFlagType _deriv_requested = {};
115
117 std::array<ConverterType, N> _converters;
118};
