267 template <typename T, typename = typename std::enable_if_t<std::is_base_of_v<Model, T>>>
268 T & register_model(const std::string & name,
269 int extra_deriv_order = 0,
270 bool nonlinear = false,
271 bool merge_input = true)
272 {
273 OptionSet extra_opts;
274 extra_opts.set<NEML2Object *>("_host") = host();
275 extra_opts.set<int>("_extra_derivative_order") = extra_deriv_order;
276 extra_opts.set<bool>("_nonlinear_system") = nonlinear;
277 extra_opts.set<bool>("_enable_AD") = input_options().get<bool>("_enable_AD");
278
279 auto model = Factory::get_object_ptr<Model>("Models", name, extra_opts);
280
281 if (merge_input)
282 for (auto && [name, var] : model->input_views())
283 declare_input_variable(var.base_storage(), var.type(), name);
284
285 _registered_models.push_back(model.get());
286 return *(std::dynamic_pointer_cast<T>(model));
287 }
