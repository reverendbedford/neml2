263 template <typename T, typename = typename std::enable_if_t<std::is_base_of_v<Model, T>>>
264 T & register_model(const std::string & name,
265 int extra_deriv_order = 0,
266 bool nonlinear = false,
267 bool merge_input = true)
268 {
269 OptionSet extra_opts;
270 extra_opts.set<NEML2Object *>("_host") = host();
271 extra_opts.set<int>("_extra_derivative_order") = extra_deriv_order;
272 extra_opts.set<bool>("_nonlinear_system") = nonlinear;
273 extra_opts.set<bool>("_inference_mode") = input_options().get<bool>("_inference_mode");
274
275 auto model = Factory::get_object_ptr<Model>("Models", name, extra_opts);
276
277 if (merge_input)
278 for (auto && [name, var] : model->input_views())
279 declare_input_variable(var.base_storage(), var.type(), name);
280
281 _registered_models.push_back(model.get());
282 return *(std::dynamic_pointer_cast<T>(model));
283 }
