141 Variable(const VariableName & name_in, TensorType type = TensorTypeEnum<T2>::value)
142 : VariableBase(name_in),
143 _type(type),
144 _base_sizes(T::const_base_sizes)
145 {
146 }
147
148 template <typename T2 = T, typename = typename std::enable_if_t<std::is_same_v<Tensor, T2>>>
149 Variable(const VariableName & name_in,
150 TensorShapeRef base_shape,
151 TensorType type = TensorType::kTensor)
152 : VariableBase(name_in),
153 _type(type),
154 _base_sizes(base_shape.vec())
155 {
156 }
157
158 virtual void setup_views(const LabeledVector * value,
159 const LabeledMatrix * deriv = nullptr,
160 const LabeledTensor3D * secderiv = nullptr) override
161 {
162 VariableBase::setup_views(value, deriv, secderiv);
163 if (value)
164 _value = T(_raw_value.view(sizes()), batch_dim());
165 }
166
167 virtual void requires_grad_(bool req = true) override { _value.requires_grad_(req); }
168
169 virtual TensorShapeRef base_sizes() const override { return _base_sizes; }
170
171 virtual TensorShapeRef sizes() const override { return _sizes; }
172
174 [[deprecated("Variable<T> must be assigned to references -- missing &")]] Variable(
175 const Variable<T> &)
176 {
177 }
178
180 [[deprecated("Variable<T> must be assigned to references -- missing &")]] void
181 operator=(const Variable<T> &)
182 {
183 }
184
191 void operator=(const Tensor & val)
192 {
193 _value.index_put_({torch::indexing::Slice()},
194 val.batch_expand(batch_sizes()).base_reshape(base_sizes()));
195 }
196
198 const T & value() const { return _value; }

201 virtual const Tensor tensor() const override { return _value; }
202
203 virtual TensorType type() const override { return _type; }

253// Everything below is just for convenience: We just forward operations to the the variable values
254// so that we can do
255//
256// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
257// var4 = (var1 - var2) * var3
258// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
259//
260// instead of the (ugly?) expression below
261//
262// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
263// var4 = (var1.v - var2.v) * var3.v
264// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
265#define FWD_VARIABLE_BINARY_OP(op) \
266 template <typename T1, \
267 typename T2, \
268 typename = typename std::enable_if_t<std::is_base_of_v<VariableBase, T1> || \
269 std::is_base_of_v<VariableBase, T2>>> \
270 auto op(const T1 & a, const T2 & b) \
271 { \
272 if constexpr (std::is_base_of_v<VariableBase, T1> && std::is_base_of_v<VariableBase, T2>) \
273 return op(a.value(), b.value()); \
274 \
275 if constexpr (std::is_base_of_v<VariableBase, T1> && !std::is_base_of_v<VariableBase, T2>) \
276 return op(a.value(), b); \
277 \
278 if constexpr (!std::is_base_of_v<VariableBase, T1> && std::is_base_of_v<VariableBase, T2>) \
279 return op(a, b.value()); \
280 } \
281 static_assert(true)
282FWD_VARIABLE_BINARY_OP(operator+);
283FWD_VARIABLE_BINARY_OP(operator-);
284FWD_VARIABLE_BINARY_OP(operator*);
285FWD_VARIABLE_BINARY_OP(operator/);
