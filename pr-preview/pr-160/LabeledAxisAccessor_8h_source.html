70 // note: The enable_if restricts Container to types other than std::string that have a .begin()
71 // and .end() that return valid input iterators.
72 template <
73 typename Container,
74 std::enable_if_t<!std::is_convertible_v<Container, std::string> &&
75 std::is_convertible_v<
76 typename std::iterator_traits<
77 decltype(std::declval<Container>().begin())>::iterator_category,
78 std::input_iterator_tag> &&
79 std::is_convertible_v<
80 typename std::iterator_traits<
81 decltype(std::declval<Container>().end())>::iterator_category,
82 std::input_iterator_tag>,
83 int> = 0>
84 LabeledAxisAccessor(Container && c)
85 {
86 _item_names.append(c.begin(), c.end());
87 for (const auto & name : _item_names)
88 validate_item_name(name);
89 }
