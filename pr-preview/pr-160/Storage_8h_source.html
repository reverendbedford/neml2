55 template <class BaseIterator>
56 struct DereferenceIterator : public BaseIterator

58 DereferenceIterator(const BaseIterator & it)

63 using key_type = typename BaseIterator::value_type::first_type;
64 using value_type = typename BaseIterator::value_type::second_type::element_type;
65
66 std::pair<key_type, value_type &> operator*() const
67 {
68 auto & [key, val] = BaseIterator::operator*();
69 neml_assert_dbg(val.get(),
70 "Trying to dereference a null object. Make sure the storage was properly "
71 "initialized using set_pointer().");
72 return {key, *val};
73 }
74 std::pair<key_type, value_type *> operator->() const

77 neml_assert_dbg(val.get(),

79 "initialized using set_pointer().");
80 return {key, val.get()};
81 }
82 };

84 using values_type = typename std::map<I, std::unique_ptr<T>>;
85 using iterator = DereferenceIterator<typename values_type::iterator>;

97 iterator begin() { return iterator(_values.begin()); }

99 const_iterator begin() const { return const_iterator(_values.begin()); }

113 T & operator[](const I & i) const
114 {
115 neml_assert_dbg(has_key(i),
116 "Trying to access a null object. Make sure the storage was properly "
117 "initialized using set_pointer().");
118 return *pointer_value(i);
119 }
120 T & operator[](const I & i) { return std::as_const(*this)[i]; }
122
129 std::size_t size() const { return _values.size(); }
130
134 bool empty() const { return _values.empty(); }
135
139 bool has_key(const I & i) const { return _values.count(i); }

148 const T * query_value(const I & i) const { return has_key(i) ? pointer_value(i).get() : nullptr; }
149 T * query_value(const I & i)

152 }
154
164 T * set_pointer(const I & i, std::unique_ptr<T> && ptr)
165 {
166 _values[i] = std::move(ptr);
167 return _values[i].get();
168 }
169
170private:
178 const std::unique_ptr<T> & pointer_value(const I & i) const { return _values.at(i); }
