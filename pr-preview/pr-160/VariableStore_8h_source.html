244 // Helper method to construct variable name in place
245 template <typename... S>
246 VariableName variable_name(S &&... name) const
247 {
248 using FirstType = std::tuple_element_t<0, std::tuple<S...>>;
249
250 if constexpr (sizeof...(name) == 1 && std::is_convertible_v<FirstType, std::string>)
251 {
252 if (_options.contains<VariableName>(name...))
253 return _options.get<VariableName>(name...);
254 return VariableName(std::forward<S>(name)...);
255 }
256 else
257 return VariableName(std::forward<S>(name)...);
258 }

260 // Create a variable view (doesn't setup the view)
261 template <typename T>
262 Variable<T> * create_variable_view(Storage<VariableName, VariableBase> & views,
263 const VariableName & name,
264 TorchSize sz = -1)
265 {
266 if constexpr (std::is_same_v<T, BatchTensor>)
267 neml_assert(sz > 0, "Allocating a BatchTensor requires a known storage size.");
268
269 // Make sure we don't duplicate variable allocation
270 VariableBase * var_base_ptr = views.query_value(name);
271 neml_assert(!var_base_ptr,
272 "Trying to allocate variable ",
273 name,
274 ", but a variable with the same name already exists.");
275
276 // Allocate
277 if constexpr (std::is_same_v<T, BatchTensor>)
278 {
279 auto var = std::make_unique<Variable<BatchTensor>>(name, sz);
280 var_base_ptr = views.set_pointer(name, std::move(var));
281 }
282 else
283 {
284 auto var = std::make_unique<Variable<T>>(name);
285 var_base_ptr = views.set_pointer(name, std::move(var));
286 }
287
288 // Cast it to the concrete type
289 auto var_ptr = dynamic_cast<Variable<T> *>(var_base_ptr);
290 neml_assert(
291 var_ptr, "Internal error: Failed to cast variable ", name, " to its concrete type.");
292
293 return var_ptr;
294 }
