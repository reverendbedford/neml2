186template <class T>
187inline std::shared_ptr<T>
188Factory::get_object_ptr(const std::string & section,
189 const std::string & name,
190 const OptionSet & additional_options,
191 bool force_create)
192{
193 auto & factory = Factory::get();
194
195 // Easy if it already exists
196 if (!force_create)
197 if (factory._objects.count(section) && factory._objects.at(section).count(name))
198 for (const auto & neml2_obj : factory._objects[section][name])
199 {
200 // Check for option clash
201 if (!options_compatible(neml2_obj->input_options(), additional_options))
202 continue;
203
204 // Check for object type
205 auto obj = std::dynamic_pointer_cast<T>(neml2_obj);
206 neml_assert(obj != nullptr,
207 "Found object named ",
208 name,
209 " under section ",
210 section,
211 ". But dynamic cast failed. Did you specify the correct object type?");
212
213 return obj;
214 }
215
216 // Otherwise try to create it
217 for (auto & options : factory._all_options[section])
218 if (options.first == name)
219 {
220 auto new_options = options.second;
221 new_options += additional_options;
222 factory.create_object(section, new_options);
223 break;
224 }
225
226 neml_assert(factory._objects.count(section) && factory._objects.at(section).count(name),
227 "Failed to get object named ",
228 name,
229 " under section ",
230 section);
231
232 auto obj = std::dynamic_pointer_cast<T>(factory._objects[section][name].back());
233 neml_assert(obj != nullptr, "Internal error: Factory failed to create object ", name);
234 return obj;
235}
