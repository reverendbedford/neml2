197template <class T>
198inline std::shared_ptr<T>
199Factory::get_object_ptr(const std::string & section,
200 const std::string & name,
201 const OptionSet & additional_options,
202 bool force_create)
203{
204 auto & factory = Factory::get();
205
206 // Easy if it already exists
207 if (!force_create)
208 if (factory._objects.count(section) && factory._objects.at(section).count(name))
209 for (const auto & neml2_obj : factory._objects[section][name])
210 {
211 // Check for option clash
212 if (!options_compatible(neml2_obj->input_options(), additional_options))
213 continue;
214
215 // Check for object type
216 auto obj = std::dynamic_pointer_cast<T>(neml2_obj);
217 neml_assert(obj != nullptr,
218 "Found object named ",
219 name,
220 " under section ",
221 section,
222 ". But dynamic cast failed. Did you specify the correct object type?");
223
224 return obj;
225 }
226
227 // Otherwise try to create it
228 for (auto & options : factory._all_options[section])
229 if (options.first == name)
230 {
231 auto new_options = options.second;
232 new_options += additional_options;
233 factory.create_object(section, new_options);
234 break;
235 }
236
237 neml_assert(factory._objects.count(section) && factory._objects.at(section).count(name),
238 "Failed to get object named ",
239 name,
240 " under section ",
241 section);
242
243 auto obj = std::dynamic_pointer_cast<T>(factory._objects[section][name].back());
244 neml_assert(obj != nullptr, "Internal error: Factory failed to create object ", name);
245 return obj;
246}
