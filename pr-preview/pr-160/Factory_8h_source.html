146template <class T>
147inline std::shared_ptr<T>
148Factory::get_object_ptr(const std::string & section,
149 const std::string & name,
150 const OptionSet & additional_options,
151 bool force_create)
152{
153 auto & factory = Factory::get();
154
155 // Easy if it already exists
156 if (!force_create)
157 if (factory._objects.count(section) && factory._objects.at(section).count(name))
158 for (const auto & neml2_obj : factory._objects[section][name])
159 {
160 // Check for option clash
161 if (!options_compatible(neml2_obj->input_options(), additional_options))
162 continue;
163
164 // Check for object type
165 auto obj = std::dynamic_pointer_cast<T>(neml2_obj);
166 neml_assert(obj != nullptr,
167 "Found object named ",
168 name,
169 " under section ",
170 section,
171 ". But dynamic cast failed. Did you specify the correct object type?");
172
173 return obj;
174 }
175
176 // Otherwise try to create it
177 for (auto & options : factory._all_options[section])
178 if (options.first == name)
179 {
180 auto new_options = options.second;
181 new_options += additional_options;
182 factory.create_object(section, new_options);
183 break;
184 }
185
186 neml_assert(factory._objects.count(section) && factory._objects.at(section).count(name),
187 "Failed to get object named ",
188 name,
189 " under section ",
190 section);
191
192 auto obj = std::dynamic_pointer_cast<T>(factory._objects[section][name].back());
193 neml_assert(obj != nullptr, "Internal error: Factory failed to create object ", name);
194 return obj;
195}
