390template <typename T>
391bool
392OptionSet::Option<T>::operator==(const OptionBase & other) const
393{
394 const auto other_ptr = dynamic_cast<const Option<T> *>(&other);
395 if (!other_ptr)
396 return false;
397
398 return (_metadata == other_ptr->_metadata) && (other_ptr->get() == this->get());

429template <typename T>
430bool
431OptionSet::contains(const std::string & name) const
432{
433 OptionSet::const_iterator it = _values.find(name);
434 if (it != _values.end())
435 if (dynamic_cast<const Option<T> *>(it->second.get()))
436 return true;
437 return false;
438}

440template <typename T>
441const T &
442OptionSet::get(const std::string & name) const
443{
444 neml_assert(this->contains<T>(name),
445 "ERROR: no option named \"",
446 name,
447 "\" found.\n\nKnown options:\n",
448 *this);
449
450 auto ptr = dynamic_cast<Option<T> *>(_values.at(name).get());
451 return ptr->get();
452}

454template <typename T>
455T &
456OptionSet::set(const std::string & name)
457{
458 if (!this->contains<T>(name))
459 _values[name] = std::make_unique<Option<T>>(name);
460 auto ptr = dynamic_cast<Option<T> *>(_values[name].get());
461 return ptr->set();
462}
