449OptionSet::Option<T>::operator==(const OptionBase & other) const
450{

484template <typename T>
485bool
486OptionSet::contains(const std::string & name) const
487{
488 OptionSet::const_iterator it = _values.find(name);
489 if (it != _values.end())
490 if (dynamic_cast<const Option<T> *>(it->second.get()))
491 return true;
492 return false;
493}

495template <typename T>
496const T &
497OptionSet::get(const std::string & name) const
498{
499 neml_assert(this->contains<T>(name),
500 "ERROR: no option named \"",
501 name,
502 "\" found.\n\nKnown options:\n",
503 *this);
504
505 auto ptr = dynamic_cast<Option<T> *>(_values.at(name).get());
506 return ptr->get();
507}

509template <typename T, FType F>
510T &
511OptionSet::set(const std::string & name)
512{
513 if (!this->contains<T>(name))
514 _values[name] = std::make_unique<Option<T>>(name);
515 auto ptr = dynamic_cast<Option<T> *>(_values[name].get());
516 ptr->ftype() = F;
517 return ptr->set();
518}
