388template <typename T>

390OptionSet::Option<T>::operator==(const OptionBase & other) const
391{
392 const auto other_ptr = dynamic_cast<const Option<T> *>(&other);
393 if (!other_ptr)

429OptionSet::contains(const std::string & name) const
430{
431 OptionSet::const_iterator it = _values.find(name);
432 if (it != _values.end())
433 if (dynamic_cast<const Option<T> *>(it->second.get()))

435 return false;

438template <typename T>
439const T &
440OptionSet::get(const std::string & name) const
441{
442 neml_assert(this->contains<T>(name),
443 "ERROR: no option named \"",
444 name,
445 "\" found.\n\nKnown options:\n",
446 *this);
447
448 auto ptr = dynamic_cast<Option<T> *>(_values.at(name).get());
449 return ptr->get();
450}

452template <typename T>
453T &
454OptionSet::set(const std::string & name)
455{
456 if (!this->contains<T>(name))
457 _values[name] = std::make_unique<Option<T>>(name);
458 auto ptr = dynamic_cast<Option<T> *>(_values[name].get());
459 return ptr->set();
460}
