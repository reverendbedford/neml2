228 template <typename T, typename = typename std::enable_if_t<std::is_base_of_v<Model, T>>>
229 T & register_model(const std::string & name,
230 int extra_deriv_order = 0,
231 bool nonlinear = false,
232 bool merge_input = true)
233 {
234 OptionSet extra_opts;
235 extra_opts.set<NEML2Object *>("_host") = host();
236 extra_opts.set<int>("_extra_derivative_order") = extra_deriv_order;
237 extra_opts.set<bool>("_nonlinear_system") = nonlinear;
238
239 auto model = Factory::get_object_ptr<Model>("Models", name, extra_opts, /*force_create=*/true);
240
241 if (merge_input)
242 for (auto && [name, var] : model->input_views())
243 declare_input_variable(var.base_storage(), name);
244
245 _registered_models.push_back(model.get());
246 return *(std::dynamic_pointer_cast<T>(model));
247 }
